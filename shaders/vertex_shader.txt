#version 460 core

layout (location = 0) in vec3 iPosition;
layout (location = 1) in vec2 iTexCoord;

out vec2 fTexCoord;

uniform mat4 uModelMat;
uniform mat4 uViewMat;
uniform mat4 uProjectionMat;

void main()
{
    vec4 vPosition;

    vPosition = vec4(iPosition, 1.0f);

    gl_Position = uProjectionMat * uViewMat * uModelMat * vPosition;
    fTexCoord = iTexCoord;
}
